    int lowbit(int x){
        return x&(-x);
    }
    void add(int x,int d,int type){
        while(x<=20000){
            c[x][type]+=d;
            x+=lowbit(x);
        }
    }
    ll sum(int x,int type){
        ll ans=0;
        while(x>0){
            ans+=c[x][type];
            x-=lowbit(x);
        }
        return ans;
    }

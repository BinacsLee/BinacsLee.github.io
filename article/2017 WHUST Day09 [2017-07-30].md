### B - String Compare HUST - 1004 
string 类型可直接用sort排序 然后对比即可 stl秒
          
          for(int i=0;i<n;i++) cin>>a[i];
          sort(a,a+n);
          int res=0;
          for(int i=0;i<n-1;i++){
              for(int j=i+1;j<n;j++){
                  if(a[i].size()>=a[j].size()) break;
                  if(a[i].compare(0,a[i].size(),a[j],0,a[i].size())==0) res++;
                  else break;
              }
          }
